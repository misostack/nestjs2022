```md
- content
  pages
    - [page_slug].[yyyymmddhhmmss].md
  posts
    - [post_slug].[yyyymmddhhmmss].md
data.json
  pages: [{slug: '', title: ''}]
  posts: [{slug: '', title: '', categories: ['','']}],
  categories: [{slug:'', title: ''}]

  siteURL:
  siteTitle:
  siteDescription:
  siteRobots:
  siteGA:
```
